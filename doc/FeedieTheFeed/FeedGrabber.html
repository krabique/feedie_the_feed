def fb_posts_limit(limit)
  unless limit.is_a?(Integer) && limit <= 100 && limit > 0
    raise BadFacebookPostsLimit, 'Facebook posts limit can only be an ' \
      'integer from 1 to 100'
  end
  @facebook_posts_limit_global = limit
end
